                int twiceButOne(int arr[],int n){
                    int c=0;
                    for(int i=0;i<n;i++){
                        c=1;
                        for(int j=0;j<n,j!=i;j++){
                            if(arr[i]==arr[j]) c++;
                        }
                        if(c==1) return arr[i];
                    }
                }
            

            

                //    bit manipulation   //

                int twiceButOne(int arr[],int n){

                    int c=0;
                    for(int i=0;i<n;i++){
                        c^=arr[i];
                    }
                    return c;
                }

                int sumSubSet(int arr[], int n, int k) {
                    int p = (1 << n) - 1; // Initialize p to represent all subsets of arr
                    for (int i = p; i > 0; i--) {
                        int s = 0;
                        int temp = i;
                        for (int j = n - 1; j >= 0; j--) {
                            s +=(temp & 1)* arr[j];
                            temp = temp >> 1;
                        }
                        if (s == k)
                            return true;
                    }
                    return false;
                }                
            

            

                //this will only the sum is present in the subsets or not

                bool sumSubSet(int arr[],int len,int target,int csum,int idx){

                    if(idx==len) 
                        return k==csum;

                    return sumSubSet(arr,len,target,csum+arr[idx],idx+1) or sumSubSet(arr,len,target,csum,idx+1)
                }

                //floor number or lesser nearest number in the sorted array
                int FloorBinarySearch(int arr[],int len,int target){
                    int lo=0,hi=len-1,ans=-1;
                    while(lo<=hi){
                        int mid=lo+(hi-lo)/2;
                        if(arr[mid]<=target){
                            ans=mid;
                            lo=mid+1;
                        }else{
                            hi=mid-1;
                        }
                    }
                    return ans?(arr[ans]==target):ans+1;
                }
